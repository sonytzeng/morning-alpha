function buildPayload(report:Record<string,unknown>): Record<string,unknown> {
  const keys=['summary','market_bias','confidence_score','confidence_label','today_quote','today_strategy','can_watch','avoid_today','fear_greed','fear_greed_summary','vix','vix_summary','risk_factors_json','watch_sectors_json','focus_stock_json','tomorrow_watch_json','global_events_json','ai_strategy_json','important_news_json','yesterday_summary','today_summary','ai_psychology','ai_retail_reminder','ai_confidence_reason','sentiment_score','sentiment_label','sentiment_reason','risk_reason'];
  const p:Record<string,unknown>={}; for (const k of keys) { if (k in report) p[k]=report[k]; }
  if (!p.ai_strategy_json||typeof p.ai_strategy_json!=='object') p.ai_strategy_json={};
  const ai={...(p.ai_strategy_json as Record<string,unknown>)};
  if (Array.isArray(report.script_valid_conditions)) ai.script_valid_conditions=report.script_valid_conditions;
  if (Array.isArray(report.script_invalid_conditions)) ai.script_invalid_conditions=report.script_invalid_conditions;
  if (report.today_key_signal) ai.today_key_signal=report.today_key_signal;
  if (report.today_key_signal_reason) ai.today_key_signal_reason=report.today_key_signal_reason;
  if (Array.isArray(report.intraday_checkpoints)) ai.intraday_checkpoints=report.intraday_checkpoints;
  p.ai_strategy_json=ai;return p;
}
